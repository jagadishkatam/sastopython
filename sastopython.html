# Define a function to implement if-elif-else logic
def calculate_merit(x):
    if x > 20:
        return "Warm"
    elif x > 10:
        return "Mild"
    else:
        return "Cold"

# Approach 1
old_data["weather"] = old_data["temp"].apply(calculate_merit)

# Approach 2
old_data["weather"] = old_data["temp"].apply(lambda x: calculate_merit(x))

new_data = old_data
